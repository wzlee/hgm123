114         }

208 /**
209 * 触发事件，注意：此接口，在外部不要调用，外部调用此函数触发的事件，为无效事件
210 * @param {String} eventType 事件类型。 
211 * @param {String} eventData 事件数据。 
212 * @private
213 */
214 IYYCommon.prototype.dispatchEvent = function(eventType, eventData) {
215     //触发事件
216     if (this.eventsMap[eventType] === null || this.eventsMap[eventType] === undefined) return;
217     for (var i = 0; i < this.eventsMap[eventType].length; i++) {
218         switch (arguments.length) {
219             case 1:
220                 this.eventsMap[eventType][i](); //不需要信息的事件
221                 break;
222             case 2:
223                 this.eventsMap[eventType][i](eventData);
224                 break;
225             default:
226         }
227     }
228 };
